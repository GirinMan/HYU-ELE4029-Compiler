void main(void)
{
	int i; int x[5];
	
	i = 0;
	while( i < 5 )
	{
		x[i] /* Comment can be anywhere */ = f(i);
		i = i + 1;
	}
	
	;/*;^_^;*/;
}

int func(int x) {
    if (x < 100)
        return x;
    else
        return func(x - 100);
}
